        $fname = $_POST['fname'];
        $lname = $_POST['lname'];
        $mail = $_POST['mail'];

        $phno = $_POST['phno'];
        $address = $_POST['address'];

        $result = mysqli_query($link, $insert);

            // Get the generated customer ID

            echo "Error: " . mysqli_error($link);
